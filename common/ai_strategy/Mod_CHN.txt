 	Dont_infantry = {
 		Enable = { 
 			OR = {
 				has_manpower < 10000
 				has_equipment = {infantry_equipment < 1000}
 				}
 			AND = {
 				has_war = yes 
 				original_tag = CHN
 				}
 				}
 		abort = {
 			AND = {
 				has_equipment = {infantry_equipment > 1500}
 				has_manpower > 25000
 				}
 				}
 	
 		ai_strategy = {
 			type = save_equipment
  		id = infantry_equipment
 			value = 500 
 			}		
 			}
